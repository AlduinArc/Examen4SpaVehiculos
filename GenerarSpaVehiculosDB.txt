-- =========================
-- creacion de las tablas
-- =========================

CREATE TABLE dbo.Ciudad (
    idCiudad INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL
);

CREATE TABLE dbo.Sede (
    idSede INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Direccion VARCHAR(200),
    idCiudad INT NOT NULL
);

CREATE TABLE dbo.Productos (
    idProducto INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Descripcion TEXT,
    Precio DECIMAL(10, 2) NOT NULL
);

CREATE TABLE dbo.Inventario (
    idInventario INT IDENTITY(1,1) PRIMARY KEY,
    idSede INT NOT NULL,
    idProducto INT NOT NULL,
    Cantidad INT NOT NULL
);

CREATE TABLE dbo.Proveedor (
    idProveedor INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Contacto VARCHAR(100)
);

CREATE TABLE dbo.DetalleProducto (
    idDetalleProducto INT IDENTITY(1,1) PRIMARY KEY,
    idProducto INT NOT NULL,
    idProveedor INT NOT NULL,
    FechaCompra DATE,
    Cantidad INT
);

CREATE TABLE dbo.DetalleVenta (
    idDetalleVenta INT IDENTITY(1,1) PRIMARY KEY,
    idProducto INT NOT NULL,
    Cantidad INT,
    PrecioUnitario DECIMAL(10, 2)
);

CREATE TABLE dbo.Servicios (
    idServicio INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100) NOT NULL,
    Descripcion TEXT,
    Precio DECIMAL(10, 2) NOT NULL,
    idSede INT NOT NULL
);

CREATE TABLE dbo.Empleado (
    idEmpleado INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    idServicio INT
);

CREATE TABLE dbo.Turno (
    idTurno INT IDENTITY(1,1) PRIMARY KEY,
    idEmpleado INT NOT NULL,
    Fecha DATE,
    HoraInicio TIME,
    HoraFin TIME
);

CREATE TABLE dbo.Cliente (
    idCliente INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    Telefono VARCHAR(20)
);

CREATE TABLE dbo.Auto (
    idAuto INT IDENTITY(1,1) PRIMARY KEY,
    idCliente INT NOT NULL,
    Marca VARCHAR(50),
    Modelo VARCHAR(50),
    Placa VARCHAR(20) UNIQUE
);

CREATE TABLE dbo.DetalleServicio (
    idDetalleServicio INT IDENTITY(1,1) PRIMARY KEY,
    idServicio INT NOT NULL,
    idAuto INT NOT NULL,
    Fecha DATE
);

CREATE TABLE dbo.Cita (
    idCita INT IDENTITY(1,1) PRIMARY KEY,
    idCliente INT NOT NULL,
    Fecha DATE,
    Hora TIME
);

CREATE TABLE dbo.Factura (
    idFactura INT IDENTITY(1,1) PRIMARY KEY,
    idCliente INT NOT NULL,
    Fecha DATE,
    Total DECIMAL(10,2)
);

CREATE TABLE dbo.DetalleFactura (
    idDetalleFactura INT IDENTITY(1,1) PRIMARY KEY,
    idFactura INT NOT NULL,
    Descripcion TEXT,
    Cantidad INT,
    PrecioUnitario DECIMAL(10, 2)
);
